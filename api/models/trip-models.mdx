## Overview

The trip models handle trip lifecycle, driver assignments, fare calculations, and location tracking for rides in the Rodando platform.

## Core Trip Models

### TripDto

Complete trip object with all details including status, location, and fare information.

**Source:** `src/app/features/trip/interfaces/trip.interface.ts:16`

```typescript
interface TripDto {
  id: string;
  currentStatus: 'pending' | 'assigning' | 'accepted' | 'arriving' | 'in_progress' | 'completed' | 'cancelled' | 'no_drivers_found';
  pickupAddress?: string | null;
  pickupPoint?: { lat: number; lng: number } | null;
  stops?: TripStopDto[];
  fareSnapshot?: FareSnapshotDto | null;
  requestedVehicleCategory?: { id: string; name?: string } | null;
  requestedServiceClass?: { id: string; name?: string } | null;
  fareEstimatedTotal?: number | null;
  fareFinalCurrency?: string | null;
  fareDistanceKm?: number | null;
  fareDurationMin?: number | null;
}
```

<ResponseField name="id" type="string" required>
  Unique identifier for the trip (UUID).
</ResponseField>

<ResponseField name="currentStatus" type="string" required>
  Current status of the trip in its lifecycle.
  
  **Possible values:**
  - `pending` - Trip created, awaiting assignment
  - `assigning` - System is finding a driver
  - `accepted` - Driver accepted the assignment
  - `arriving` - Driver is en route to pickup location
  - `in_progress` - Trip is in progress
  - `completed` - Trip finished successfully
  - `cancelled` - Trip was cancelled
  - `no_drivers_found` - No available drivers found
</ResponseField>

<ResponseField name="pickupAddress" type="string" optional>
  Human-readable pickup address (e.g., "123 Main St, New York, NY").
</ResponseField>

<ResponseField name="pickupPoint" type="object" optional>
  Geographic coordinates of the pickup location.
  
  <Expandable title="Coordinates">
    <ResponseField name="lat" type="number" required>
      Latitude coordinate
    </ResponseField>
    <ResponseField name="lng" type="number" required>
      Longitude coordinate
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="stops" type="TripStopDto[]" optional>
  Array of intermediate stops and final destination.
  
  <Expandable title="TripStopDto properties">
    <ResponseField name="seq" type="number" required>
      Sequence number for the stop (0-based index)
    </ResponseField>
    <ResponseField name="point" type="object" optional>
      Geographic coordinates of the stop
      
      <Expandable title="Coordinates">
        <ResponseField name="lat" type="number" required>
          Latitude coordinate
        </ResponseField>
        <ResponseField name="lng" type="number" required>
          Longitude coordinate
        </ResponseField>
      </Expandable>
    </ResponseField>
    <ResponseField name="address" type="string" optional>
      Human-readable address for the stop
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="fareSnapshot" type="FareSnapshotDto" optional>
  Snapshot of fare calculation at trip creation time.
  
  <Expandable title="FareSnapshotDto properties">
    <ResponseField name="currency" type="string" required>
      Currency code (e.g., "USD", "EUR")
    </ResponseField>
    <ResponseField name="surgeMultiplier" type="number" optional>
      Surge pricing multiplier (e.g., 1.5 for 1.5x pricing)
    </ResponseField>
    <ResponseField name="breakdown" type="any" optional>
      Detailed fare breakdown object
    </ResponseField>
    <ResponseField name="distanceKmEst" type="number" optional>
      Estimated distance in kilometers
    </ResponseField>
    <ResponseField name="durationMinEst" type="number" optional>
      Estimated duration in minutes
    </ResponseField>
    <ResponseField name="estimatedTotal" type="number" optional>
      Estimated total fare amount
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="requestedVehicleCategory" type="object" optional>
  The type of vehicle requested for this trip.
  
  <Expandable title="Vehicle category">
    <ResponseField name="id" type="string" required>
      Vehicle category ID
    </ResponseField>
    <ResponseField name="name" type="string" optional>
      Vehicle category name (e.g., "Sedan", "SUV")
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="requestedServiceClass" type="object" optional>
  The service level requested for this trip.
  
  <Expandable title="Service class">
    <ResponseField name="id" type="string" required>
      Service class ID
    </ResponseField>
    <ResponseField name="name" type="string" optional>
      Service class name (e.g., "Economy", "Premium")
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="fareEstimatedTotal" type="number" optional>
  Estimated total fare for the trip.
</ResponseField>

<ResponseField name="fareFinalCurrency" type="string" optional>
  Currency for the final fare (e.g., "USD").
</ResponseField>

<ResponseField name="fareDistanceKm" type="number" optional>
  Distance used for fare calculation in kilometers.
</ResponseField>

<ResponseField name="fareDurationMin" type="number" optional>

### FareSnapshotDto

Snapshot of fare calculation parameters at the time of trip creation.

**Source:** `src/app/features/trip/interfaces/trip.interface.ts:7`

```typescript
interface FareSnapshotDto {
  currency: string;
  surgeMultiplier?: number | null;
  breakdown?: any;
  distanceKmEst?: number | null;
  durationMinEst?: number | null;
  estimatedTotal?: number | null;
}
```

<ResponseField name="currency" type="string" required>
  ISO 4217 currency code (e.g., "USD", "EUR", "GBP").
</ResponseField>

<ResponseField name="surgeMultiplier" type="number" optional>
  Surge pricing multiplier applied to base fare. A value of 1.5 means 1.5x the base price.
</ResponseField>

<ResponseField name="breakdown" type="any" optional>
  Detailed breakdown of fare components. Structure varies based on implementation.
</ResponseField>

<ResponseField name="distanceKmEst" type="number" optional>
  Estimated trip distance in kilometers.
</ResponseField>

<ResponseField name="durationMinEst" type="number" optional>
  Estimated trip duration in minutes.
</ResponseField>

<ResponseField name="estimatedTotal" type="number" optional>
  Total estimated fare amount in the specified currency.
</ResponseField>

<CodeGroup>
```json Example Fare Snapshot
{
  "currency": "USD",
  "surgeMultiplier": 1.3,
  "breakdown": {
    "baseFare": 5.00,
    "distanceFare": 12.50,
    "timeFare": 3.00,
    "surgeFee": 6.15,
    "serviceFee": 2.50
  },
  "distanceKmEst": 8.2,
  "durationMinEst": 22,
  "estimatedTotal": 29.15
}
```
</CodeGroup>
